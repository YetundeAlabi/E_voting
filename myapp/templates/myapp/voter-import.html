// Get the file input element var fileInput = document.getElementById("file-input"); // Create a new FormData object var formData = new FormData(); // Append the file to the FormData object formData.append("file", fileInput.files[0]); // Make the POST request with the FormData object fetch("http://your-api-endpoint/", { method: "POST", body: formData }) .then(function(response) { if (response.ok) { // File uploaded successfully console.log("File uploaded successfully!"); } else { // Error in uploading file console.log("Failed to upload file."); } }) .catch(function(error) { // Error in making the request console.log("An error occurred while uploading the file."); console.log(error); });
